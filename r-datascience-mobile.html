3.8 if / for / while / switch syntax

# if-else
if (x > 0) "pos" else if (x==0) "zero" else "neg"

# for
for (i in 1:5) print(i^2)

# while
i <- 1
while (i <= 5) { print(i); i <- i+1 }

# repeat
i <- 1
repeat { if (i>5) break; print(i); i <- i+1 }

# next (skip)
for (i in 1:5) { if (i==3) next; print(i) }

# switch
day <- "mon"
switch(day, mon="Start", fri="TGIF", "Mid-week")

3.12 OOP — S3 vs S4 ⭐ classic 8M

S3 — informal

x <- list(a=1)
class(x) <- "student"
print.student <- function(o) cat("Student:", o$a)
print(x)            # dispatches to print.student

S4 — formal

setClass("Student",
  representation(name="character", marks="numeric"))
s <- new("Student", name="A", marks=85)

setGeneric("show2", function(x) standardGeneric("show2"))
setMethod("show2", "Student", function(x) cat(x@name, x@marks))
show2(s)            # "A 85"

getMode <- function(v) {
  u <- unique(v)
  u[which.max(tabulate(match(v, u)))]

P4. Grading using for-loop + if-else ⭐

marks <- c(85, 67, 45, 92, 30, 78)
grade <- character(length(marks))

for (i in 1:length(marks)) {
  if (marks[i] >= 75) {
    grade[i] <- "A"
  } else if (marks[i] >= 60) {
    grade[i] <- "B"
  } else if (marks[i] >= 50) {
    grade[i] <- "C"
  } else {
    grade[i] <- "F"
  }
}

result <- data.frame(marks, grade)
print(result)
#   marks grade
# 1    85     A
# 2    67     B
# 3    45     F
# 4    92     A
# 5    30     F
# 6    78     A
8M scoring template Sample data → init grade vec → for + nested if-else → bind df → print.

P5. UDF + if-else + for combined

checkEvenOdd <- function(nums) {
  for (n in nums) {
    if (n %% 2 == 0) {
      cat(n, "is Even\n")
    } else {
      cat(n, "is Odd\n")
    }
  }
}

checkEvenOdd(c(2, 7, 10, 15))

# 7 is Odd
# 10 is Even
# 15 is Odd
